## Interactions

### 1. Update Organization Info
1. User modifies fields
2. Save button becomes enabled
3. Clicks "Save Changes"
4. System validates input
5. Shows success message
6. Updates organization settings

### 2. Manage Billing
1. User clicks "Manage Subscription"
2. Opens billing portal
3. Can view/update payment method
4. Can change plan
5. Can view invoices

### 3. View Usage Metrics
1. User views usage dashboard
2. Sees real-time usage data
3. Can drill down for details
4. Receives warnings at 80% and 90%

### 4. Toggle Features
1. User checks/unchecks feature flag
2. System validates plan permissions
3. Enables/disables feature
4. Shows confirmation

### 5. Export Settings
1. User clicks export option
2. Selects export format
3. Downloads settings backup
4. Can import to restore

// Validate before save
function validateSettings(settings: OrganizationSettings) {
  const errors = [];

  if (!settings.name || settings.name.length < 2) {
    errors.push('Organization name must be at least 2 characters');
  }

  if (!isValidEmail(settings.contactEmail)) {
    errors.push('Valid email address is required');
  }

  return errors;
}
