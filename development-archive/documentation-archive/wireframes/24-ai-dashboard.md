### Metrics Collection
```typescript
interface AIMetrics {
  timestamp: Date;
  service: string;
  model: string;
  requestCount: number;
  successCount: number;
  errorCount: number;
  avgResponseTime: number;
  tokensUsed: number;
  qualityScore: number;
}

async function collectMetrics() {
  const metrics = await Promise.all([
    getServiceMetrics('document-analysis'),
    getServiceMetrics('document-generation'),
    getServiceMetrics('quality-scoring'),
  ]);

  // Calculate aggregates
  const aggregated = {
    totalRequests: sum(metrics.map(m => m.requestCount)),
    totalTokens: sum(metrics.map(m => m.tokensUsed)),
    avgSuccessRate: avg(metrics.map(m => m.successCount / m.requestCount)),
    avgResponseTime: avg(metrics.map(m => m.avgResponseTime)),
    avgQualityScore: avg(metrics.map(m => m.qualityScore)),
  };

  return aggregated;
}
```
